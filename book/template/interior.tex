% ===== Interior design (xelatex / TeX Live only) — loaded for the PDF build =====
% Fonts: TeX Gyre family ships with texlive-fonts-recommended (installed in CI).
\usepackage{fontspec}
\setmainfont{TeX Gyre Pagella}      % body — Palatino-like, warm and readable at 10pt
\setsansfont{TeX Gyre Heros}        % headings/labels — Helvetica-like
\setmonofont{TeX Gyre Cursor}[Scale=0.86]   % code

\usepackage{microtype}

\usepackage{xcolor}
\definecolor{brand}{HTML}{1C5D99}     % restrained brand blue
\definecolor{brandink}{HTML}{14304A}  % near-black ink for headings
\definecolor{codebg}{HTML}{F5F7F9}
\definecolor{codeframe}{HTML}{D5DCE2}

% ---- headings (sans, brand accent) ----
\usepackage{titlesec}
\titleformat{\chapter}[display]
  {\normalfont\sffamily\bfseries\color{brand}}
  {\raggedleft\fontsize{58}{58}\selectfont\color{brand!25}\thechapter}
  {-4pt}
  {\Huge\raggedright}
\titlespacing*{\chapter}{0pt}{14pt}{26pt}
\titleformat{\section}{\sffamily\Large\bfseries\color{brandink}}{\thesection}{0.55em}{}
\titleformat{\subsection}{\sffamily\large\bfseries\color{brandink}}{\thesubsection}{0.55em}{}
\titleformat{\subsubsection}{\sffamily\normalsize\bfseries\color{brandink}}{}{0em}{}

% ---- running heads + folios ----
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\renewcommand{\headrulewidth}{0.3pt}
\fancyhead[LE]{\small\textsf{\thepage}\hspace{1em}\small\textsf{\nouppercase{\leftmark}}}
\fancyhead[RO]{\small\textsf{\nouppercase{\rightmark}}\hspace{1em}\small\textsf{\thepage}}
\renewcommand{\chaptermark}[1]{\markboth{#1}{}}
\renewcommand{\sectionmark}[1]{\markright{#1}}
\fancypagestyle{plain}{\fancyhf{}\renewcommand{\headrulewidth}{0pt}\fancyfoot[C]{\small\textsf{\thepage}}}

% ---- links in brand colour ----
\hypersetup{colorlinks=true,linkcolor=brandink,urlcolor=brand,citecolor=brand}

% ---- code blocks: box every verbatim block ----
\usepackage{mdframed}
\mdfdefinestyle{codebox}{backgroundcolor=codebg,linecolor=codeframe,linewidth=0.4pt,
  innerleftmargin=8pt,innerrightmargin=8pt,innertopmargin=6pt,innerbottommargin=6pt,
  skipabove=9pt,skipbelow=9pt,roundcorner=2pt}
\surroundwithmdframed[style=codebox]{verbatim}

% ---- editorial callouts: pandoc blockquotes get an accent left rule ----
\newmdenv[topline=false,bottomline=false,rightline=false,linewidth=2pt,
  linecolor=brand!60,leftmargin=0pt,innerleftmargin=10pt,innerrightmargin=0pt,
  innertopmargin=1pt,innerbottommargin=1pt,skipabove=8pt,skipbelow=8pt]{quotebar}
\renewenvironment{quote}{\begin{quotebar}\small}{\end{quotebar}}
